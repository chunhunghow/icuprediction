-- Find patient base

SELECT distinct
HSP.PAT_ID
, HSP.PAT_ENC_CSN_ID
, HSP.INPATIENT_DATA_ID
, ZC_SEX.NAME												AS GENDER
, DATEDIFF(MONTH, PAT.BIRTH_DATE, HSP.INP_ADM_DATE)	/ 12	AS AGE_AT_ADMISSION
, HSP.INP_ADM_DATE
, HSP.ADT_PAT_CLASS_C
, DEP.DEPARTMENT_NAME
, DISPO.NAME												AS DISCHARGE_DISPOSITION
INTO #BASE
FROM PAT_ENC_HSP HSP
INNER JOIN PATIENT PAT ON HSP.PAT_ID = PAT.PAT_ID
INNER JOIN CLARITY_DEP DEP ON HSP.DEPARTMENT_ID = DEP.DEPARTMENT_ID
INNER JOIN X_V_NUHS_ENCOUNTERS ON HSP.PAT_ENC_CSN_ID = X_V_NUHS_ENCOUNTERS.PAT_ENC_CSN_ID
LEFT OUTER JOIN ZC_SEX ON PAT.SEX_C = ZC_SEX.RCPT_MEM_SEX_C
LEFT OUTER JOIN ZC_DISCH_DISP DISPO ON HSP.DISCH_DISP_C = DISPO.DISCH_DISP_C
LEFT OUTER JOIN V_PATIENT_EXCLUSION ON HSP.PAT_ID = V_PATIENT_EXCLUSION.PAT_ID
WHERE HSP.INP_ADM_DATE >= '2016-01-01'
AND HSP.INP_ADM_DATE < '2021-01-01'
AND HSP.HOSPITAL_AREA_ID = '10101'									-- SS : 10101- Ng Teng Fong General Hospital
AND DATEDIFF(MONTH, PAT.BIRTH_DATE, HSP.INP_ADM_DATE) / 12 >= 18	-- SS: Patients 18 years and above
AND HSP.ADT_PAT_CLASS_C IN ('101', '103')							-- SS : 101 - Inpatient, 103 - Emergency
AND V_PATIENT_EXCLUSION.PAT_ID IS NULL								-- SS : Excluding the SAP and FDR testing patients
ORDER BY HSP.PAT_ID, HSP.PAT_ENC_CSN_ID

-- Find required flowsheet value of patient base (Assumption: MEAS.FLO_MEAS_ID = '3047610003' is the only relevant flowsheet ID for this request)

SELECT DISTINCT
#BASE.PAT_ENC_CSN_ID
, MEAS.RECORDED_TIME
, MEAS.MEAS_VALUE
INTO #MV
FROM #BASE
LEFT OUTER JOIN IP_FLWSHT_REC REC ON #BASE.INPATIENT_DATA_ID = REC.INPATIENT_DATA_ID
LEFT OUTER JOIN IP_FLWSHT_MEAS MEAS ON REC.FSD_ID = MEAS.FSD_ID
LEFT OUTER JOIN IP_FLO_GP_DATA FLO ON MEAS.FLO_MEAS_ID = FLO.FLO_MEAS_ID
WHERE MEAS.FLO_MEAS_ID = '3047610003'
ORDER BY #BASE.PAT_ENC_CSN_ID

-- Required column extraction

SELECT DISTINCT
#BASE.PAT_ID
, #BASE.PAT_ENC_CSN_ID
, #BASE.GENDER
, #BASE.INP_ADM_DATE
, #BASE.AGE_AT_ADMISSION
, #BASE.DEPARTMENT_NAME
, #BASE.DISCHARGE_DISPOSITION
--, #MV.RECORDED_TIME
, #MV.MEAS_VALUE
INTO #TEMP_MV_PATIENTS
FROM #BASE
LEFT OUTER JOIN #MV ON #BASE.PAT_ENC_CSN_ID = #MV.PAT_ENC_CSN_ID
WHERE #MV.MEAS_VALUE IS NOT NULL
ORDER BY #MV.MEAS_VALUE, #BASE.INP_ADM_DATE, #BASE.PAT_ID, #BASE.PAT_ENC_CSN_ID

select distinct
	 ORDER_MED.PAT_ENC_CSN_ID as [Encounter ID]
	,ORDER_MED.PAT_ID as [Patient ID]
	,ORDER_MED.DESCRIPTION as [Order Description]
	,CLARITY_MEDICATION.NAME as [Medication Name]
	,CLARITY_MEDICATION.GENERIC_NAME as [Medication Generic Name]
	,MAR_ADMIN_INFO.TAKEN_TIME as [Time of Administration]
	,MAR_ADMIN_INFO.SIG as [Dose Amount]
	,DOSE.NAME as [Dose Unit]
	,MAR_ADMIN_INFO.INFUSION_RATE as [Infusion Rate]
	,INFUSION.NAME as [Infusion Unit]
	,MAR_ADMIN_INFO.MAR_DURATION as [Administration Duration]
	,ZC_MED_DURATION_UN.NAME as [Duration Unit]
	--,*
from 
	#TEMP_MV_PATIENTS
left join ORDER_MED on ORDER_MED.PAT_ENC_CSN_ID = #TEMP_MV_PATIENTS.PAT_ENC_CSN_ID
inner join CLARITY_MEDICATION on ORDER_MED.MEDICATION_ID = CLARITY_MEDICATION.MEDICATION_ID
inner join MAR_ADMIN_INFO on ORDER_MED.ORDER_MED_ID = MAR_ADMIN_INFO.ORDER_MED_ID
left join ZC_MED_UNIT INFUSION on INFUSION.DISP_QTYUNIT_C = MAR_ADMIN_INFO.MAR_INF_RATE_UNIT_C
left join ZC_MED_UNIT DOSE on DOSE.DISP_QTYUNIT_C = MAR_ADMIN_INFO.DOSE_UNIT_C
left join ZC_MED_DURATION_UN on ZC_MED_DURATION_UN.MED_DURATION_UN_C = MAR_ADMIN_INFO.MAR_DURATION_UNIT_C

where 1=1
and (CLARITY_MEDICATION.name like '%noradrenaline%' 
or CLARITY_MEDICATION.name like '%adrenaline%'
or CLARITY_MEDICATION.name like '%dopamine%'
or CLARITY_MEDICATION.name like '%dobutamine%'
or CLARITY_MEDICATION.name like '%phenylephrine%'
or CLARITY_MEDICATION.name like '%epinephrine%'
or CLARITY_MEDICATION.name like '%norepinephrine%')
and TAKEN_TIME is not null
and MAR_ACTION_C = 1




drop table #TEMP_MV_PATIENTS


DROP TABLE
#BASE
, #MV
